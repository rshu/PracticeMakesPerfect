628. Maximum Product of Three Numbers

Given an integer array nums, find three numbers whose product is maximum and return the maximum product.

Example 1:

Input: nums = [1,2,3]
Output: 6
Example 2:

Input: nums = [1,2,3,4]
Output: 24
Example 3:

Input: nums = [-1,-2,-3]
Output: -6


Constraints:

3 <= nums.length <= 104
-1000 <= nums[i] <= 1000

```java
class Solution {
    public int maximumProduct(int[] nums) {

        Arrays.sort(nums);
        int len = nums.length;

        // 如果数组中全是非负数，则排序后最大的三个数相乘即为最大乘积
        // 如果全是非正数，则最大的三个数相乘同样也为最大乘积
        int max1 = nums[len-1]*nums[len-2]*nums[len-3];

        // 如果数组中有正数有负数，
        // 则最大乘积既可能是三个最大正数的乘积
        // 也可能是两个最小负数（即绝对值最大）与最大正数的乘积
        int max2 = nums[0]*nums[1]*nums[len-1];
        
        return Math.max(max1, max2);
    }
}
```

```java
class Solution {
    public int maximumProduct(int[] nums) {

        Arrays.sort(nums);
        int len = nums.length;

        // 如果数组中全是非负数，则排序后最大的三个数相乘即为最大乘积
        // 如果全是非正数，则最大的三个数相乘同样也为最大乘积
       int max1 = Integer.MIN_VALUE;
       int max2 = Integer.MIN_VALUE;
       int max3 = Integer.MIN_VALUE;

       int min1 = Integer.MAX_VALUE;
       int min2 = Integer.MAX_VALUE;

       for (int num : nums) {
           if (num < min1) {
               min2 = min1;
               min1 = num;
           } else if (num < min2) {
               min2 = num;
           }

           if (num > max1) {
               max3 = max2;
               max2 = max1;
               max1 = num;
           } else if (num > max2) {
               max3 = max2;
               max2 = num;
           } else if (num > max3) {
               max3  = num;
           }
       }

        // 如果数组中有正数有负数，
        // 则最大乘积既可能是三个最大正数的乘积
        // 也可能是两个最小负数（即绝对值最大）与最大正数的乘积
        int res1 = max1 * max2 * max3;
        int res2 = min1 * min2 * max1;
        
        return Math.max(res1, res2);
    }
}
```

```java
class Solution {
    public int maximumProduct(int[] nums) {

        Arrays.sort(nums);
        int len = nums.length;

        // 如果数组中全是非负数，则排序后最大的三个数相乘即为最大乘积
        // 如果全是非正数，则最大的三个数相乘同样也为最大乘积

        // smallest top
        PriorityQueue<Integer> pq1 = new PriorityQueue<>(
            (o1, o2) -> o1 - o2
        );

        // largest top
        PriorityQueue<Integer> pq2 = new PriorityQueue<>(
            (o1, o2) -> o2 - o1
        );

        for (int num : nums) {
            pq1.offer(num);
            pq2.offer(num);

            if (pq1.size() > 3) {
                pq1.poll();
            }

            if (pq2.size() > 2) {
                pq2.poll();
            }
        }

        int max3 = pq1.poll();
        int max2 = pq1.poll();
        int max1 = pq1.poll();

        int min2 = pq2.poll();
        int min1 = pq2.poll();

        // 如果数组中有正数有负数，
        // 则最大乘积既可能是三个最大正数的乘积
        // 也可能是两个最小负数（即绝对值最大）与最大正数的乘积
        int res1 = max1 * max2 * max3;
        int res2 = min1 * min2 * max1;
        
        return Math.max(res1, res2);
    }
}
```
